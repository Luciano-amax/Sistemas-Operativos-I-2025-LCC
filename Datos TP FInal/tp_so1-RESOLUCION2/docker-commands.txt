docker pull erlang:26-alpine

docker build -t erlang-download-server .

docker network create erlang-net

docker run -d -ti --name node1 --network erlang-net \
  -p 5000:12345 -p 5001:12346 \
  erlang-download-server

docker exec -it node1 sh

docker run -d -ti --name node2 --network erlang-net \
  -p 5002:12345 -p 5003:12346 \
  erlang-download-server

docker exec -it node2 sh
